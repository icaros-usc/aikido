286 template <typename _Scalar, int _Dim, int _Degree>
287 Eigen::DenseIndex BSpline<_Scalar, _Dim, _Degree>::Span(
288  typename SplineTraits<BSpline<_Scalar, _Dim, _Degree> >::Scalar u,
289  Eigen::DenseIndex degree,
290  const typename SplineTraits<BSpline<_Scalar, _Dim, _Degree> >::
291  KnotVectorType& knots)
292 {
293  // Piegl & Tiller, "The NURBS Book", A2.1 (p. 68)
294  if (u <= knots(0))
295  return degree;
296  const Scalar* pos = std::upper_bound(
297  knots.data() + degree - 1, knots.data() + knots.size() - degree - 1, u);
298  return static_cast<Eigen::DenseIndex>(std::distance(knots.data(), pos) - 1);
299 }

301 template <typename _Scalar, int _Dim, int _Degree>
302 typename BSpline<_Scalar, _Dim, _Degree>::BasisVectorType
303 BSpline<_Scalar, _Dim, _Degree>::BasisFunctions(
304  typename BSpline<_Scalar, _Dim, _Degree>::Scalar u,
305  Eigen::DenseIndex degree,
306  const typename BSpline<_Scalar, _Dim, _Degree>::KnotVectorType& knots)
307 {
308  typedef
309  typename BSpline<_Scalar, _Dim, _Degree>::BasisVectorType BasisVectorType;
310 
311  const Eigen::DenseIndex p = degree;
312  const Eigen::DenseIndex i = BSpline::Span(u, degree, knots);
313 
314  const KnotVectorType& U = knots;
315 
316  BasisVectorType left(p + 1);
317  left(0) = Scalar(0);
318  BasisVectorType right(p + 1);
319  right(0) = Scalar(0);
320 
321  Eigen::VectorBlock<BasisVectorType, Degree>(left, 1, p)
322  = u
323  - Eigen::VectorBlock<const KnotVectorType, Degree>(U, i + 1 - p, p)
324  .reverse();
325  Eigen::VectorBlock<BasisVectorType, Degree>(right, 1, p)
326  = Eigen::VectorBlock<const KnotVectorType, Degree>(U, i + 1, p) - u;
327 
328  BasisVectorType N(1, p + 1);
329  N(0) = Scalar(1);
330  for (Eigen::DenseIndex j = 1; j <= p; ++j)
331  {
332  Scalar saved = Scalar(0);
333  for (Eigen::DenseIndex r = 0; r < j; r++)
334  {
335  const Scalar tmp = N(r) / (right(r + 1) + left(j - r));
336  N[r] = saved + right(r + 1) * tmp;
337  saved = left(j - r) * tmp;
338  }
339  N(j) = saved;
340  }
341  return N;
342 }

382 template <typename SplineType, typename DerivativeType>
383 void derivativesImpl(
384  const SplineType& spline,
385  typename SplineType::Scalar u,
386  Eigen::DenseIndex order,
387  DerivativeType& der)
388 {
389  enum
390  {
391  Dimension = SplineTraits<SplineType>::Dimension
392  };
393  enum
394  {
395  Order = SplineTraits<SplineType>::OrderAtCompileTime
396  };
397  enum
398  {
399  DerivativeOrder = DerivativeType::ColsAtCompileTime
400  };
401 
402  typedef typename SplineTraits<SplineType>::ControlPointVectorType
403  ControlPointVectorType;
404  typedef
405  typename SplineTraits<SplineType, DerivativeOrder>::BasisDerivativeType
406  BasisDerivativeType;
407  typedef typename BasisDerivativeType::ConstRowXpr BasisDerivativeRowXpr;
408 
409  const Eigen::DenseIndex p = spline.degree();
410  const Eigen::DenseIndex span = spline.span(u);
411 
412  const Eigen::DenseIndex n = (std::min)(p, order);
413 
414  der.resize(Dimension, n + 1);
415 
416  // Retrieve the basis function derivatives up to the desired order...
417  const BasisDerivativeType basis_func_ders
418  = spline.template basisFunctionDerivatives<DerivativeOrder>(u, n + 1);
419 
420  // ... and perform the linear combinations of the control points.
421  for (Eigen::DenseIndex der_order = 0; der_order < n + 1; ++der_order)
422  {
423  const Eigen::Replicate<BasisDerivativeRowXpr, Dimension, 1> ctrl_weights(
424  basis_func_ders.row(der_order));
425  const Eigen::Block<const ControlPointVectorType, Dimension, Order> ctrl_pts(
426  spline.ctrls(), 0, span - p, Dimension, p + 1);
427  der.col(der_order) = (ctrl_weights * ctrl_pts).rowwise().sum();
428  }
429 }

463 template <typename _Scalar, int _Dim, int _Degree>
464 template <typename DerivativeType>
465 void BSpline<_Scalar, _Dim, _Degree>::BasisFunctionDerivativesImpl(
466  const typename BSpline<_Scalar, _Dim, _Degree>::Scalar u,
467  const Eigen::DenseIndex order,
468  const Eigen::DenseIndex p,
469  const typename BSpline<_Scalar, _Dim, _Degree>::KnotVectorType& U,
470  DerivativeType& N_)
471 {
472  typedef BSpline<_Scalar, _Dim, _Degree> SplineType;
473  enum
474  {
475  Order = SplineTraits<SplineType>::OrderAtCompileTime
476  };
477 
478  typedef typename SplineTraits<SplineType>::Scalar Scalar;
479  typedef typename SplineTraits<SplineType>::BasisVectorType BasisVectorType;
480 
481  const Eigen::DenseIndex span = SplineType::Span(u, p, U);
482 
483  const Eigen::DenseIndex n = (std::min)(p, order);
484 
485  N_.resize(n + 1, p + 1);
486 
487  BasisVectorType left = BasisVectorType::Zero(p + 1);
488  BasisVectorType right = BasisVectorType::Zero(p + 1);
489 
490  Eigen::Matrix<Scalar, Order, Order> ndu(p + 1, p + 1);
491 
492  Scalar saved, temp; // FIXME These were double instead of Scalar. Was there a
493  // reason for that?
494 
495  ndu(0, 0) = 1.0;
496 
497  Eigen::DenseIndex j;
498  for (j = 1; j <= p; ++j)
499  {
500  left[j] = u - U[span + 1 - j];
501  right[j] = U[span + j] - u;
502  saved = 0.0;
503 
504  for (Eigen::DenseIndex r = 0; r < j; ++r)
505  {
506  /* Lower triangle */
507  ndu(j, r) = right[r + 1] + left[j - r];
508  temp = ndu(r, j - 1) / ndu(j, r);
509  /* Upper triangle */
510  ndu(r, j) = static_cast<Scalar>(saved + right[r + 1] * temp);
511  saved = left[j - r] * temp;
512  }
513 
514  ndu(j, j) = static_cast<Scalar>(saved);
515  }
516 
517  for (j = p; j >= 0; --j)
518  N_(0, j) = ndu(j, p);
519 
520  // Compute the derivatives
521  DerivativeType a(n + 1, p + 1);
522  Eigen::DenseIndex r = 0;
523  for (; r <= p; ++r)
524  {
525  Eigen::DenseIndex s1, s2;
526  s1 = 0;
527  s2 = 1; // alternate rows in array a
528  a(0, 0) = 1.0;
529 
530  // Compute the k-th derivative
531  for (Eigen::DenseIndex k = 1; k <= static_cast<Eigen::DenseIndex>(n); ++k)
532  {
533  Scalar d = 0.0;
534  Eigen::DenseIndex rk, pk, j1, j2;
535  rk = r - k;
536  pk = p - k;
537 
538  if (r >= k)
539  {
540  a(s2, 0) = a(s1, 0) / ndu(pk + 1, rk);
541  d = a(s2, 0) * ndu(rk, pk);
542  }
543 
544  if (rk >= -1)
545  j1 = 1;
546  else
547  j1 = -rk;
548 
549  if (r - 1 <= pk)
550  j2 = k - 1;
551  else
552  j2 = p - r;
553 
554  for (j = j1; j <= j2; ++j)
555  {
556  a(s2, j) = (a(s1, j) - a(s1, j - 1)) / ndu(pk + 1, rk + j);
557  d += a(s2, j) * ndu(rk + j, pk);
558  }
559 
560  if (r <= pk)
561  {
562  a(s2, k) = -a(s1, k - 1) / ndu(pk + 1, r);
563  d += a(s2, k) * ndu(r, pk);
564  }
565 
566  N_(k, r) = static_cast<Scalar>(d);
567  j = s1;
568  s1 = s2;
569  s2 = j; // Switch rows
570  }
571  }
572 
573  /* Multiply through by the correct factors */
574  /* (Eq. [2.9]) */
575  r = p;
576  for (Eigen::DenseIndex k = 1; k <= static_cast<Eigen::DenseIndex>(n); ++k)
577  {
578  for (j = p; j >= 0; --j)
579  N_(k, j) *= r;
580  r *= p - k;
581  }
582 }
